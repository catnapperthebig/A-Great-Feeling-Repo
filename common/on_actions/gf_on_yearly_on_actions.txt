		effect = {
			if = {
				limit = {
					has_army_experience > 175
				}
				set_temp_variable = { sev_lexp = army_experience }
				subtract_from_temp_variable = { sev_lexp = 175 }
				multiply_temp_variable = { sev_lexp = -1 }

		}
	}
